Executor Framework in Java

1. Understanding Executor Framework

Pseudocode: Executor Framework Concept


// Pseudocode for Executor Framework Management

class ExecutorFramework:
    task_queue = create_concurrent_queue()
    thread_pool = create_thread_pool()
    
    function submit_task(task):
        // Decouples task submission from execution
        task_queue.add(task)
        
        if available_thread_in_pool():
            execute_next_task_from_queue()
    
    function execute_task(task):
        try:
            result = task.run()
            handle_task_success(result)
        catch Exception:
            handle_task_failure()
        finally:
            mark_thread_as_available()
    
    function shutdown():
        stop_accepting_new_tasks()
        wait_for_existing_tasks_to_complete()
        terminate_thread_pool()

Key Executor Framework Principles

  • Separate task submission from task execution
  • Manage thread lifecycle automatically
  • Provide flexible task management
  • Simplify concurrent programming

2. Real-World Scenario: Data Processing Pipeline

Problem Scenario

  • Large-scale data processing
  • Multiple transformation stages
  • Need for parallel and efficient processing

Executor Framework Solution

  • Use Callable for complex tasks
  • Manage task execution and results
  • Handle exceptions gracefully

import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.*;

// Data Processing Task
class DataProcessingTask implements Callable {
    private RawData rawData;
    
    public DataProcessingTask(RawData rawData) {
        this.rawData = rawData;
    }
    
    @Override
    public ProcessedData call() throws Exception {
        // Simulate complex data processing
        return processData(rawData);
    }
    
    private ProcessedData processData(RawData rawData) throws InterruptedException {
        // Simulate multiple processing stages
        Thread.sleep((long) (Math.random() * 1000)); // Simulate processing time
        
        return new ProcessedData(
            rawData.getId(),
            rawData.getValue().toUpperCase(),
            calculateComplexity(rawData)
        );
    }
    
    private int calculateComplexity(RawData rawData) {
        return rawData.getValue().length();
    }
}

// Data Model Classes
class RawData {
    private String id;
    private String value;
    
    public RawData(String id, String value) {
        this.id = id;
        this.value = value;
    }
    
    public String getId() { return id; }
    public String getValue() { return value; }
}

class ProcessedData {
    private String id;
    private String processedValue;
    private int complexity;
    
    public ProcessedData(String id, String processedValue, int complexity) {
        this.id = id;
        this.processedValue = processedValue;
        this.complexity = complexity;
    }
    
    @Override
    public String toString() {
        return "ProcessedData{" +
               "id='" + id + '\'' +
               ", processedValue='" + processedValue + '\'' +
               ", complexity=" + complexity +
               '}';
    }
}

// Data Processing Pipeline
public class DataProcessingPipeline {
    public static void main(String[] args) {
        // Create a thread pool with 4 threads
        ExecutorService executor = Executors.newFixedThreadPool(4);
        
        // Prepare raw data
        List rawDataList = generateSampleData();
        
        // List to collect processing results
        List> processingResults = new ArrayList<>();
        
        // Submit processing tasks
        for (RawData rawData : rawDataList) {
            DataProcessingTask task = new DataProcessingTask(rawData);
            Future future = executor.submit(task);
            processingResults.add(future);
        }
        
        // Collect and process results
        for (Future result : processingResults) {
            try {
                ProcessedData processedData = result.get();
                System.out.println("Processed: " + processedData);
            } catch (InterruptedException | ExecutionException e) {
                e.printStackTrace();
            }
        }
        
        // Shutdown the executor
        executor.shutdown();
        
        try {
            // Wait for all tasks to complete
            if (!executor.awaitTermination(5, TimeUnit.SECONDS)) {
                executor.shutdownNow();
            }
        } catch (InterruptedException e) {
            executor.shutdownNow();
        }
    }
    
    // Generate sample data for processing
    private static List generateSampleData() {
        List rawDataList = new ArrayList<>();
        rawDataList.add(new RawData("1", "hello"));
        rawDataList.add(new RawData("2", "world"));
        rawDataList.add(new RawData("3", "java"));
        rawDataList.add(new RawData("4", "executor"));
        rawDataList.add(new RawData("5", "framework"));
        return rawDataList;
    }
}
