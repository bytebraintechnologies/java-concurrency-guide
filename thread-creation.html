1. Understanding Thread Creation

Pseudocode: Thread Creation Concepts


// Pseudocode for Thread Creation Patterns

// Method 1: Inheritance-based Thread Creation
class InheritanceThread extends BaseThread:
    override run_method():
        // Define thread's core logic
        execute_specific_task()

// Method 2: Interface-based Thread Creation
class InterfaceThread implements Runnable:
    run_method():
        // Define thread's core logic
        execute_specific_task()

// Method 3: Anonymous Class Creation
create_thread_with_anonymous_class():
    new Thread(new Runnable():
        run_method():
            // Inline thread logic
            execute_task_directly()
    ).start()

// Method 4: Lambda-based Thread Creation (Modern Java)
create_thread_with_lambda():
    new Thread(() -> {
        // Compact thread logic
        execute_task_directly()
    }).start()

Key Thread Creation Principles

  • Define thread's primary task/logic
  • Choose appropriate creation method
  • Understand thread lifecycle
  • Manage thread resources efficiently

import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.ExecutorService;
import java.util.concurrent.Executors;
import javax.imageio.ImageIO;

// Image Processing Demonstration
public class ParallelImageProcessor {
    // Method 1: Thread Inheritance Approach
    static class ImageProcessingThread extends Thread {
        private BufferedImage image;
        
        public ImageProcessingThread(BufferedImage image) {
            this.image = image;
        }
        
        @Override
        public void run() {
            applyGrayscaleFilter(image);
        }
        
        private void applyGrayscaleFilter(BufferedImage image) {
            for (int y = 0; y < image.getHeight(); y++) {
                for (int x = 0; x < image.getWidth(); x++) {
                    int rgb = image.getRGB(x, y);
                    int r = (rgb >> 16) & 0xFF;
                    int g = (rgb >> 8) & 0xFF;
                    int b = rgb & 0xFF;
                    
                    int gray = (r + g + b) / 3;
                    int grayRGB = (gray << 16) | (gray << 8) | gray;
                    
                    image.setRGB(x, y, grayRGB);
                }
            }
            System.out.println("Processed image in thread: " + Thread.currentThread().getName());
        }
    }
    
    // Method 2: Runnable Interface Approach
    static class ImageProcessingRunnable implements Runnable {
        private BufferedImage image;
        
        public ImageProcessingRunnable(BufferedImage image) {
            this.image = image;
        }
        
        @Override
        public void run() {
            applyBrightnessFilter(image);
        }
        
        private void applyBrightnessFilter(BufferedImage image) {
            for (int y = 0; y < image.getHeight(); y++) {
                for (int x = 0; x < image.getWidth(); x++) {
                    int rgb = image.getRGB(x, y);
                    int r = Math.min(255, ((rgb >> 16) & 0xFF) + 50);
                    int g = Math.min(255, ((rgb >> 8) & 0xFF) + 50);
                    int b = Math.min(255, (rgb & 0xFF) + 50);
                    
                    int brightenedRGB = (r << 16) | (g << 8) | b;
                    image.setRGB(x, y, brightenedRGB);
                }
            }
            System.out.println("Processed image in thread: " + Thread.currentThread().getName());
        }
    }
    
    public static void main(String[] args) {
        try {
            // Load sample images
            List images = loadImages();
            
            // Method 1: Thread Inheritance Approach
            List inheritanceThreads = new ArrayList<>();
            for (BufferedImage image : images) {
                ImageProcessingThread thread = new ImageProcessingThread(image);
                thread.start();
                inheritanceThreads.add(thread);
            }
            
            // Wait for inheritance threads to complete
            for (Thread thread : inheritanceThreads) {
                thread.join();
            }
            
            // Method 2: Runnable Interface with ExecutorService
            ExecutorService executor = Executors.newFixedThreadPool(4);
            for (BufferedImage image : images) {
                executor.submit(new ImageProcessingRunnable(image));
            }
            
            // Shutdown executor
            executor.shutdown();
            executor.awaitTermination(1, java.util.concurrent.TimeUnit.MINUTES);
            
            System.out.println("All image processing completed!");
            
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
    
    // Utility method to load sample images
    private static List loadImages() {
        List images = new ArrayList<>();
        try {
            // Replace with actual image loading logic
            BufferedImage sampleImage = new BufferedImage(100, 100, BufferedImage.TYPE_INT_RGB);
            images.add(sampleImage);
            images.add(sampleImage);
            images.add(sampleImage);
        } catch (Exception e) {
            e.printStackTrace();
        }
        return images;
    }
}
